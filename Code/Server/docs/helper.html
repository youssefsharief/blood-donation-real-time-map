function findObjects(array, property, id) {
    var results = [];
    for (var i = 0; i < array.length; i++) {
        if (id == 'null' || id == '') {
            results.push(array[i])
            continue;
        }
        var test = array[i][property];
        if (test && test.toLowerCase().indexOf(id.toLowerCase()) != -1) {
            results.push(array[i]);
        }
    }
    return results;
}
